CREATE TABLE IF NOT EXISTS fish_rus (
	id INT PRIMARY KEY AUTO_INCREMENT,
	class INT,
	subclass INT,
	groups INT,
	family INT,
	genus INT,
	species INT,
	subspecies INT,
	specimen INT,
	FOREIGN KEY (class) REFERENCES class_fish_rus (id),
	FOREIGN KEY (subclass) REFERENCES subclass_fish_rus (id),
	FOREIGN KEY (groups) REFERENCES groups_fish_rus (id),
	FOREIGN KEY (family) REFERENCES family_fish_rus (id),
	FOREIGN KEY (genus) REFERENCES genus_fish_rus (id),
	FOREIGN KEY (species) REFERENCES species_fish_rus (id),
	FOREIGN KEY (subspecies) REFERENCES subspecies_fish_rus (id),
	FOREIGN KEY (specimen) REFERENCES specimen_fish_rus (id),
	image BLOB
);
