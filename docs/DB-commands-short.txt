CREATE DATABASE fishatlas;

USE fishatlas;

CREATE TABLE class_fish (
	id INT PRIMARY KEY,
	name VARCHAR(1024) NOT NULL,
	discription VARCHAR(2048)
);

CREATE TABLE groups_fish (
	id INT PRIMARY KEY,
	name VARCHAR(1024) NOT NULL,
	discription VARCHAR(2048)
);

	id INT PRIMARY KEY AUTO_INCREMENT,

	name VARCHAR(1024) NOT NULL,
	image VARCHAR(1024),
